#include <iostream>
#include <vector>

long long fibonacciIterativo(int n) {
    if (n <= 0) return 1; 
    if (n == 1) return 2; 

    long long a = 1;
    long long b = 2; 
    long long c = 0;

    for (int i = 2; i <= n; i++) {
        c = a + b;
        a = b;
        b = c;
    }
    return b;
}

int main() {
    int n;
    std::cout << "Insert the value of n: ";
    std::cin >> n;
    std::cout << "Fibonacci(" << n << ") = " << fibonacciIterativo(n) << std::endl;
    return 0;
}
